  wrap.className='polaroid-wrap';
  const left=60+i*(clW-200)/4;
  wrap.innerHTML=`
    <div class="polaroid-clip"></div>
    <div class="polaroid" style="--rot:${ph.rot}deg;">
      <div class="polaroid-img" style="background-image:url('${ph.img}');"></div>
      <div class="polaroid-caption">${ph.caption}</div>
    </div>`;
  wrap.style.setProperty('--sdelay', `${i * 0.5}s`);
  wrap.style.setProperty('--rot', `${ph.rot}deg`);
  wrap.style.left = left + 'px';
  wrap.style.top = '20px';

  const polaroid = wrap.querySelector('.polaroid');

  // 3D Tilt Logic
  wrap.addEventListener('mousemove', e => {
    if (dragging) return;
    const rect = wrap.getBoundingClientRect();
    const x = e.clientX - rect.left;
    const y = e.clientY - rect.top;
    const centerX = rect.width / 2;
    const centerY = rect.height / 2;
    const rotateX = ((y - centerY) / centerY) * -15; // Max 15 deg
    const rotateY = ((x - centerX) / centerX) * 15;
    polaroid.style.transform = `rotateX(${rotateX}deg) rotateY(${rotateY}deg) rotateZ(var(--rot)) scale(1.05)`;
  });

  wrap.addEventListener('mouseleave', () => {
    polaroid.style.transform = '';
  });

  let dragging=false,startX=0,startY=0,origLeft=left,origTop=20,curX=left,curY=20,swingAngle=0;

  function getXY(e){
    if(e.touches)return{x:e.touches[0].clientX,y:e.touches[0].clientY};
    return{x:e.clientX,y:e.clientY};
  }

  wrap.addEventListener('mousedown',e=>{dragging=true;const{x,y}=getXY(e);startX=x-curX;startY=y-curY;wrap.style.zIndex=100;e.preventDefault();});
  wrap.addEventListener('touchstart',e=>{dragging=true;const{x,y}=getXY(e);startX=x-curX;startY=y-curY;wrap.style.zIndex=100;},{passive:true});

  function onMove(e){
    if(!dragging)return;
    const{x,y}=getXY(e);
    curX=x-startX;curY=y-startY;
    swingAngle=(curX-origLeft)*0.05;
    wrap.style.left=curX+'px';
    wrap.style.top=curY+'px';
    wrap.style.transform=`rotate(${swingAngle}deg)`;
  }

  function onUp(){
    if(!dragging)return;
    dragging=false;
    wrap.style.zIndex='';
    const spring=setInterval(()=>{
      curX+=(origLeft-curX)*0.15;
      curY+=(origTop-curY)*0.15;
      swingAngle*=0.8;
      wrap.style.left=curX+'px';
      wrap.style.top=curY+'px';
      wrap.style.transform=`rotate(${swingAngle}deg)`;
      if(Math.abs(curX-origLeft)<0.5&&Math.abs(curY-origTop)<0.5){
        curX=origLeft;curY=origTop;
        wrap.style.left=origLeft+'px';
        wrap.style.top=origTop+'px';
        wrap.style.transform='';
        clearInterval(spring);
      }
    },16);
  }

  document.addEventListener('mousemove',onMove);
  document.addEventListener('mouseup',onUp);
  document.addEventListener('touchmove',onMove,{passive:true});
  document.addEventListener('touchend',onUp);

  cl.appendChild(wrap);
});

  entries.forEach(en=>{
    if(en.isIntersecting) {
      if(!en.target.classList.contains('visible-active')) {
        // Prepare starting point
        en.target.style.transform = isScrollingDown ? 'translateX(-80px)' : 'translateX(80px)';
        en.target.style.opacity = '0';
        en.target.style.transition = 'none'; // Disable transition for instant snap

        en.target.offsetHeight; 
        
        // Start animation
        requestAnimationFrame(() => {
          en.target.style.transition = ''; // Restore transition
          en.target.classList.add('visible-active');
          // Clear inline styles so they don't block CSS
          en.target.style.transform = '';
          en.target.style.opacity = '';
        });
      }
    } else {
      en.target.classList.remove('visible-active');
    }
  });
},{threshold: 0.1});

const canvas = document.getElementById('audio-visualizer');
const ctx = canvas.getContext('2d');
let cw, ch;

function resizeVisualizer() {
  cw = canvas.width = window.innerWidth;
  ch = canvas.height = 120;
}
window.addEventListener('resize', resizeVisualizer);
resizeVisualizer();

let vTime = 0;
function animateVisualizer() {
  ctx.clearRect(0, 0, cw, ch);
  vTime += 0.015;
  
  const bars = 80;
  const barGap = 2;
  const barWidth = (cw / bars) - barGap;
  
  for(let i=0; i<bars; i++) {
    const hue = 200 + Math.sin(vTime + i*0.05) * 20;
    // Procedural movement simulating audio frequencies
    const slow = Math.sin(vTime + i * 0.15) * 25;
    const fast = Math.sin(vTime * 2.5 + i * 0.4) * 10;
    const peak = Math.random() < 0.05 ? Math.random() * 20 : 0; // Random peaks
    
    const height = 15 + Math.abs(slow + fast) + peak;
    
    ctx.fillStyle = `hsla(${hue}, 100%, 70%, 0.3)`;
    ctx.fillRect(i * (cw/bars), ch, barWidth, -height);
    
    // Glossy top
    ctx.fillStyle = `hsla(${hue}, 100%, 80%, 0.6)`;
    ctx.fillRect(i * (cw/bars), ch - height, barWidth, 2);
  }
  requestAnimationFrame(animateVisualizer);
}
animateVisualizer();

/* ‚îÄ‚îÄ‚îÄ DYNAMIC SCROLL FOCUS ‚îÄ‚îÄ‚îÄ */
const focusSections = document.querySelectorAll('section, footer, .wali-card');
function updateScrollFocus() {
  const vh = window.innerHeight;
  const centerY = vh / 2;

  focusSections.forEach(sec => {
    const rect = sec.getBoundingClientRect();
    const secCenter = rect.top + rect.height / 2;
    const dist = Math.abs(centerY - secCenter);
    
    // Lower intensity and wider threshold for better visibility
    const threshold = vh * 0.35; // Wider sharp zone
    const maxDist = vh * 0.8;
    
    let intensity = 0;
    if (dist > threshold) {
      intensity = Math.min((dist - threshold) / (maxDist - threshold), 1);
    }
    
    // Cap blur at 5px instead of 10px, and min opacity at 0.5
    const blur = (intensity * 5).toFixed(1);
    const opacity = (1 - intensity * 0.5).toFixed(2);
    
    sec.style.filter = `blur(${blur}px)`;
    sec.style.opacity = opacity;
  });
}

window.addEventListener('scroll', () => {
  requestAnimationFrame(updateScrollFocus);
});
updateScrollFocus(); // Initial call
